CREATE
VIEW t2tierp.VIEW_SPED_NFE_ITEM
AS

select 
	distinct(P.ID),
	P.NOME,
	P.GTIN,
	P.ID_UNIDADE_PRODUTO,
	U.SIGLA,
	P.TIPO_ITEM_SPED,
	P.NCM,
	P.EX_TIPI,
	P.CODIGO_LST,
	P.ALIQUOTA_ICMS_PAF
from
	PRODUTO P 
	INNER JOIN NFE_DETALHE NFED ON (NFED.ID_PRODUTO = P.ID)
	INNER JOIN UNIDADE_PRODUTO U ON (P.ID_UNIDADE_PRODUTO = U.ID)
