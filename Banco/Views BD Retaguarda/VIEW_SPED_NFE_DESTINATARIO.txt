CREATE
VIEW t2tierp.VIEW_SPED_NFE_DESTINATARIO
AS

select 
	DISTINCT(NFE.ID_CLIENTE) AS ID,
	E.NOME,
	E.CPF_CNPJ,
	E.INSCRICAO_ESTADUAL,
	E.CODIGO_MUNICIPIO,
	E.SUFRAMA,
	E.LOGRADOURO,
	E.NUMERO,
	E.COMPLEMENTO,
	E.BAIRRO
from
	NFE_CABECALHO NFE 
	INNER JOIN NFE_DESTINATARIO E ON (E.ID_NFE_CABECALHO = NFE.ID)
where 
	NFE.ID_CLIENTE IS NOT NULL
